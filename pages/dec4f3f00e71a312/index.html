# LeetCode 刷题

func maxArea(height []int) int {
    // height 是一个装有高度值的数组
    max, start, end := 0, 0, len(height) - 1
    for start < end {
        width := end - start
        high := 0
        if height[start] < height[end] {
            high = height[start]
            start++
        } else {
            high = height[end]
            end--
        }
        temp := width * high
		if temp > max {
			max = temp
		}
    }
    return max
}
